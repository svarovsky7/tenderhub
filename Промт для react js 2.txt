SELECT
  tables.table_name,
  columns.column_name,
  columns.data_type,
  columns.is_nullable,
  columns.column_default
FROM
  information_schema.tables
JOIN
  information_schema.columns
ON
  tables.table_name = columns.table_name
WHERE
  tables.table_schema = 'public'
ORDER BY
  tables.table_name,
  columns.ordinal_position;
